FUNCIONES DE REGISTRO

assert() - APARECE UN MENSAJE DE ERROR EN LA CONSOLA SI LA AFIRMACION ES FALSA. SI LA AFIRMACION ES VERDADERA, NO APARECERA NADA (NO ESTANDAR, NO RECOMENDABLE)

clear() - LIMPIA LA CONSOLA 

error() - MUESTRA UN MENSAJE DE ERROR EN LA CONSOLA

info() - EMITE UN MENSAJE INFORMATIVO A LA CONSOLA WEB, EN FIREFOX Y CHROME, SE MUESTRE UN ICONO PEQUEÃ‘O "I" JUNTO A ESTOS ELEMENTOS EN EL REGISTRO DE LA CONSOLA WEB

log() - MUESTRA UN MENSAJE EN LA CONSOLA WEB (O DEL INTERPRETE JAVASCRIPT)

table() - ESTA FUNCION TOMA UN ARGUMENTO OBLIGATORIO DATA, DQUE DEBE SER UN ARRAY O UN OBJETO, Y UN PARAMETRO ADICCIONAL: COLUMNAS Y NOS MUESTRA UNA TABLA EN LA CONSOLA

warn() - IMPRIME UN MENSAJE DE ADVERTENCIA EN LA CONSOLA WEB

dir() - DESPLIEGA UNA LISTA INTERACTIVA DE LAS PROPIEDADES DEL OBJETO JAVASCRIPT ESPECIFICADO (NO ESTANDAR, NO RECOMENDABLE)

FUNCIONES DE CONTEO

count() - REGISTRA EL NUMERO DE VECES QUE SE LLAMA A COUNT(). ESTA FUNCION TOMA COMO ARGUMENTO OPCIONAL UNA ETIQUETA.

countReset() - RESETEA EL CONTADOR console.count();

FUNCIONES DE AGRUPACION

group() - CREA UN NUEVO GRUPO EN LINEA EN EL REGISTRO DE LA CONSOLA WEB

groupEnd() - REMUEVE UN GRUPO EN LINEA EN EL REGISTRO DE LA CONSOLA WEB

groupCollapsed() - CREA UN NUEVO GRUPO EN LINEA PERO CONTRAIDO, EL USUARIO DEBE EXPANDIRLO PARA VERLO

FUNCIONES DE TEMPORIZACION

time() - INICIA UN TEMPORIZADOR

timeEnd() - REGISTRA EL VALOR ACTUAL DE UN TEMPORIZADOR

timeLog() - DETIENE UN TEMPORIZADOR
